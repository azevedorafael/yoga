### Usage

```javascript state
render(() => {
  const [value, setValue] = useState('');

  return (
    <Input.Email
      label="E-mail"
      helper="Insert a valid e-mail address"
      value={value}
      onChange={e => setValue(e.target.value)}
      onClean={cleaned => setValue(cleaned)}
    />
  );
});
```
