  display: flex;
  flex-direction: column;
`;

  <svg {...props}>
    <circle cx="6" cy="7" r="5" />
  </svg>
);

render(
  <Center>
    <Rating icon={Circle} value={1} />
    <Rating icon={Circle} value={1.5} />
