### Reference

Gympass `<PlanCard />` component is used to display content related to an user's
plan.

### Usage

#### Simple

```javascript
<PlanCard
  ribbon={{
    text: 'Tag Label',
  }}
>
  <PlanCard.Content
    title="Gympass Black for employees with Buddha Spa + Reebook included"
    price="US$99.90"
    period="/month"
  />

  <PlanCard.Actions>
    <Button.Link variant="secondary">See gyms included</Button.Link>
  </PlanCard.Actions>
</PlanCard>
```

#### Ribbon variant

```javascript
<PlanCard
  ribbon={{
    text: 'Medium Label',
    variant: 'secondary',
  }}
>
  <PlanCard.Content
    title="Gympass Black for employees with Buddha Spa + Reebook included"
    price="US$129.90"
    period="/month"
  />

  <PlanCard.Actions>
    <Button.Link variant="secondary">See gyms included</Button.Link>
  </PlanCard.Actions>
</PlanCard>
```

#### PlanCard variant

```javascript
<PlanCard
  ribbon={{
    text: 'Super large label in french',
  }}
  variant="secondary"
>
  <PlanCard.Content
    title="Gympass Black for employees with Buddha Spa + Reebook included"
    price="US$299.90"
    period="/month"
  />
  <PlanCard.Actions>
    <Button.Link inverted>See gyms included</Button.Link>
  </PlanCard.Actions>
</PlanCard>
```
