### Reference

Gympass `<Progress />` description

### Usage

```javascript type=expo
<ScrollView
  style={{
    width: '100%',
    height: '100%',
    padding: 80,
  }}
>
  <Text>Default</Text>
  <Progress max={100} value={35} />

  <Text>Numeric value on the left (default)</Text>
  <Progress max={100} value={20} label={{ value: 20 }} />
  <Progress max={100} value={40} label={{ value: 40 }} />
  <Progress max={100} value={67} label={{ value: 67 }} />

  <Text>Numeric value on the right</Text>
  <Progress max={100} value={100} label={{ value: 100, placement: 'right' }} />
  <Progress max={100} value={65} label={{ value: 65, placement: 'right' }} />
  <Progress max={100} value={12} label={{ value: 12, placement: 'right' }} />

  <Text>Alphabetic value on the left (default)</Text>
  <Progress max={100} value={30} label={{ value: 'Some description' }} />

  <Text>Alphabetic value on the right (default)</Text>
  <Progress
    max={100}
    value={70}
    label={{ value: 'Some description', placement: 'right' }}
  />
</ScrollView>
```
