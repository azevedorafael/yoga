### Reference

Gympass `<List />` component implements all our design guidelines, be sure to wrap your items with `<List.Item/>` that uses default spacing, border and color pallete tokens.

### Usage

```javascript reactLive=true state=true
const SwitchLine = styled.div`
  display: flex;
  justify-content: space-between;
`;

render(() => {
  const [locker, setLocker] = useState(false);
  const [wifi, setWifi] = useState(false);
  const [parking, setParking] = useState(false);

  return (
    <List>
      <List.Item>
        <SwitchLine>
          <span>Locker</span>
          <Checkbox.Switch
            checked={locker}
            onChange={() => setLocker(!locker)}
          />
        </SwitchLine>
      </List.Item>

      <List.Item>
        <SwitchLine>
          <span>Wi-fi</span>
          <Checkbox.Switch checked={wifi} onChange={() => setWifi(!wifi)} />
        </SwitchLine>
      </List.Item>

      <List.Item>
        <SwitchLine>
          <span>Parking</span>
          <Checkbox.Switch
            checked={parking}
            onChange={() => setParking(!parking)}
          />
        </SwitchLine>
      </List.Item>
    </List>
  );
});
```
