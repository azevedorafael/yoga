### Reference

`<RadioGroup.Button />` creates a button instead of a radio default looks, be sure to use `<RadioGroup.Button />` as `<RadioGroup />` children so it inherit default spacing, border and color pallete tokens.

### Usage

```javascript type=expo state
const { useState } = React;

const App = () => {
  const [selectedValue, setSelectedValue] = useState('24h');
  const [category, setCategory] = useState('Gyms');

  return (
    <View>
      <Text>RadioGroup.Button</Text>

      <RadioGroup
        selectedValue={selectedValue}
        onChange={({ value }) => setSelectedValue(value)}
      >
        <RadioGroup.Button value="24h">24 hours</RadioGroup.Button>
        <RadioGroup.Button value="now">Open now</RadioGroup.Button>
        <RadioGroup.Button value="sundays">At Sundays</RadioGroup.Button>
      </RadioGroup>

      <Text>RadioGroup.Radio</Text>

      <RadioGroup
        selectedValue={category}
        onChange={({ value }) => setCategory(value)}
        style={{
          paddingBottom: 40,
        }}
      >
        <RadioGroup.Radio
          style={{
            marginRight: 4,
          }}
          value="Gyms"
        />
        <RadioGroup.Radio value="Classes" />
      </RadioGroup>
    </View>
  );
};

export default App;
```
