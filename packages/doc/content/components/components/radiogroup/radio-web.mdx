### Usage

```javascript state
render(() => {
  const [selectedValue, setSelectedValue] = useState('now');

  return (
    <RadioGroup
      name="openning"
      selectedValue={selectedValue}
      onChange={({ target: { value } }) => setSelectedValue(value)}
    >
      <RadioGroup.Radio style={{ marginRight: 4 }} value="24h">
        24 hours
      </RadioGroup.Radio>
      <RadioGroup.Radio style={{ marginRight: 4 }} value="now">
        Open now
      </RadioGroup.Radio>
      <RadioGroup.Radio value="sundays">At Sundays</RadioGroup.Radio>
    </RadioGroup>
  );
});
```
